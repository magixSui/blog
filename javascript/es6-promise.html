实例化/初始化promise

function initializePromise(promise, resolver) {
// 传入了当前对象 预处理的函数
  try {
  // 只是传入函数作为参数，还没有执行
    resolver(function resolvePromise(value) {
      resolve(promise, value);
    }, function rejectPromise(reason) {
      reject(promise, reason);
    });
  } catch (e) {
    reject(promise, e);
  }
}
